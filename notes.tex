\section{Control Flow}

    \begin{lstlisting}
        (*fn*) gcd(a:(`int`), b:(`int`)):(`int`)
        {
            (*if*) {
                a==0 : 0
                b==0 : 0
                (*else*) : 
                (*do*) {
                    a<0 : a=-a
                    b<0 : b=-b
                }; (*do*) {
                    a<b : b=mod(b,a)
                    a>b : a=mod(a,b)
                }; a
            }
        }
    \end{lstlisting}
